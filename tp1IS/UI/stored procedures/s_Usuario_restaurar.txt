create PROCEDURE [dbo].[s_Usuario_restaurar](
	@id int,
	@password nvarchar(50),
	@username nvarchar(50),
	@birthdate nvarchar(50),
	@active bit
	)

AS SET NOCOUNT ON

	UPDATE Usuario SET password = @password, username = @username, active = @active, birthdate = @birthdate where id=@id
	exec s_Usuario_historico_crear @id= @id, @user= @username, @password=@password, @active=@active, @birthdate=@birthdate
