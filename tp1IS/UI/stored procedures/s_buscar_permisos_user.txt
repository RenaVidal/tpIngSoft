create PROCEDURE [dbo].[s_buscar_permisos_user](
	@id numeric(18,0)
	)

AS SET NOCOUNT ON
SELECT t1.*
FROM permiso t1
JOIN permiso_usuario t2 ON t1.id = t2.id_rol
WHERE t2.id_usuario = @id
