create PROCEDURE [dbo].[s_Usuario_login](
	@username  nvarchar(50),
	@password  nvarchar(50)
)

AS SET NOCOUNT ON

	SELECT  * FROM dbo.Usuario 
WHERE username = @username AND password = @password
